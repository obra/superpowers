# Appendix B: Dedicated SDS Review Interface Implementation

## Overview

The Appendix B Dedicated SDS Review Interface provides a specialized, HITL-enabled workspace for safety officers to review and approve Safety Data Sheets (SDS) generated by AI agents. This interface follows the **Page Implementation Procedure** as a simple page with integrated chatbot and HITL capabilities.

**Page Classification**: Simple Page (Template A - Clean)
**Route**: `/appendix-b-sds-review/:taskId`
**Purpose**: Human-in-the-Loop review of AI-generated Safety Data Sheets with specialized compliance tools

## Core SDS Review Interface Component

### SDSReviewInterface.js
```javascript
import React, { useState, useEffect } from 'react';
import SafetyComplianceTools from './SafetyComplianceTools';
import HITLChatInterface from './HITLChatInterface';
import RevisionRequestModal from './modals/RevisionRequestModal';

const SDSReviewInterface = ({ sdsData, taskId, onRevisionRequest, onApproval }) => {
  const [expandedSections, setExpandedSections] = useState(new Set([1])); // Section 1 expanded by default
  const [sectionValidations, setSectionValidations] = useState({});
  const [showRevisionModal, setShowRevisionModal] = useState(false);
  const [selectedSection, setSelectedSection] = useState(null);

  const toggleSection = (sectionId) => {
    const newExpanded = new Set(expandedSections);
    if (newExpanded.has(sectionId)) {
      newExpanded.delete(sectionId);
    } else {
      newExpanded.add(sectionId);
    }
    setExpandedSections(newExpanded);
  };

  const handleSectionValidation = (sectionId, isValid, notes) => {
    setSectionValidations(prev => ({
      ...prev,
      [sectionId]: { isValid, notes, timestamp: new Date() }
    }));
  };

  const requestRevision = (sectionId) => {
    setSelectedSection(sectionId);
    setShowRevisionModal(true);
  };

  const submitRevisionRequest = async (revisionData) => {
    await onRevisionRequest(selectedSection, revisionData);
    setShowRevisionModal(false);
    setSelectedSection(null);
  };

  const submitApproval = async () => {
    const allSectionsValid = Object.values(sectionValidations).every(v => v.isValid);
    if (allSectionsValid) {
      await onApproval(sectionValidations);
    } else {
      alert('Please validate all sections before approving');
    }
  };

  return (
    <div className="sds-review-container">
      <div className="sds-content-grid">
        {/* SDS Content Panel */}
        <div className="sds-sections-panel">
          <h3>Safety Data Sheet Content</h3>

          {sdsData.sections.map(section => (
            <SDSSection
              key={section.id}
              section={section}
              isExpanded={expandedSections.has(section.id)}
              validation={sectionValidations[section.id]}
              onToggle={() => toggleSection(section.id)}
              onValidate={(isValid, notes) => handleSectionValidation(section.id, isValid, notes)}
              onRequestRevision={() => requestRevision(section.id)}
            />
          ))}

          <div className="sds-actions" style={{ marginTop: '20px', paddingTop: '20px', borderTop: '1px solid #e5e7eb' }}>
            <button
              onClick={() => setShowRevisionModal(true)}
              className="btn btn-warning"
              style={{ marginRight: '10px' }}
            >
              Request Revision
            </button>
            <button
              onClick={submitApproval}
              className="btn btn-success"
            >
              Approve SDS
            </button>
          </div>
        </div>

        {/* Compliance Tools Panel */}
        <div className="compliance-tools-panel">
          <SafetyComplianceTools
            sdsData={sdsData}
            onValidationUpdate={handleSectionValidation}
          />
        </div>
      </div>

      {/* HITL Chat Interface */}
      <div className="safety-chat-container">
        <HITLChatInterface
          taskId={taskId}
          context="sds_review"
          agentType="safety_compliance"
        />
      </div>

      {/* Revision Request Modal */}
      {showRevisionModal && (
        <RevisionRequestModal
          section={selectedSection ? sdsData.sections.find(s => s.id === selectedSection) : null}
          onSubmit={submitRevisionRequest}
          onClose={() => setShowRevisionModal(false)}
        />
      )}
    </div>
  );
};

export default SDSReviewInterface;
```

import React, { useState } from 'react';

const SDSSection = ({
  section,
  isExpanded,
  validation,
  onToggle,
  onValidate,
  onRequestRevision
}) => {
  const [isValid, setIsValid] = useState(validation?.isValid || false);
  const [validationNotes, setValidationNotes] = useState(validation?.notes || '');

  const handleValidationChange = (valid) => {
    setIsValid(valid);
    onValidate(valid, validationNotes);
  };

  const getConfidenceClass = (confidence) => {
    if (confidence >= 0.8) return 'confidence-high';
    if (confidence >= 0.6) return 'confidence-medium';
    return 'confidence-low';
  };

  const getConfidenceIcon = (confidence) => {
    if (confidence >= 0.8) return '✓';
    if (confidence >= 0.6) return '⚠️';
    return '❌';
  };

  return (
    <div className="sds-section" style={{ marginBottom: '10px', border: '1px solid #e5e7eb', borderRadius: '6px' }}>
      <div
        className="sds-section-header"
        onClick={onToggle}
        style={{
          padding: '12px 16px',
          background: '#f9fafb',
          borderBottom: isExpanded ? '1px solid #e5e7eb' : 'none',
          cursor: 'pointer',
          borderRadius: isExpanded ? '6px 6px 0 0' : '6px',
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center'
        }}
      >
        <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
          <span style={{ fontSize: '18px' }}>
            {isExpanded ? '▼' : '▶'}
          </span>
          <span style={{ fontWeight: '600' }}>
            Section {section.id}: {section.title}
          </span>
          <span className={`confidence-indicator ${getConfidenceClass(section.aiConfidence)}`}>
            {getConfidenceIcon(section.aiConfidence)} {Math.round(section.aiConfidence * 100)}%
          </span>
        </div>

        {validation && (
          <span style={{
            color: validation.isValid ? '#059669' : '#dc2626',
            fontSize: '14px',
            fontWeight: '500'
          }}>
            {validation.isValid ? '✓ Validated' : '⚠ Needs Review'}
          </span>
        )}
      </div>

      {isExpanded && (
        <div className="sds-section-content" style={{ padding: '16px' }}>
          <div className="sds-section-text" style={{ marginBottom: '16px' }}>
            {section.content}
          </div>

          {section.needsReview && (
            <div className="review-notice" style={{
              background: '#fef3c7',
              border: '1px solid #f59e0b',
              borderRadius: '4px',
              padding: '8px 12px',
              marginBottom: '16px',
              color: '#92400e'
            }}>
              ⚠️ AI flagged this section for human review: {section.reviewReason}
            </div>
          )}

          <div className="section-validation" style={{ marginTop: '16px' }}>
            <label style={{ display: 'block', marginBottom: '8px', fontWeight: '500' }}>
              Section Validation:
            </label>

            <div style={{ display: 'flex', gap: '12px', marginBottom: '12px' }}>
              <label style={{ display: 'flex', alignItems: 'center', gap: '6px' }}>
                <input
                  type="radio"
                  name={`validation-${section.id}`}
                  checked={isValid === true}
                  onChange={() => handleValidationChange(true)}
                />
                Valid
              </label>

              <label style={{ display: 'flex', alignItems: 'center', gap: '6px' }}>
                <input
                  type="radio"
                  name={`validation-${section.id}`}
                  checked={isValid === false}
                  onChange={() => handleValidationChange(false)}
                />
                Needs Revision
              </label>
            </div>

            <textarea
              placeholder="Validation notes or revision requirements..."
              value={validationNotes}
              onChange={(e) => setValidationNotes(e.target.value)}
              onBlur={() => onValidate(isValid, validationNotes)}
              style={{
                width: '100%',
                minHeight: '60px',
                padding: '8px',
                border: '1px solid #d1d5db',
                borderRadius: '4px',
                resize: 'vertical'
              }}
            />

            {!isValid && (
              <button
                onClick={onRequestRevision}
                className="btn btn-sm btn-warning"
                style={{ marginTop: '8px' }}
              >
                Request AI Revision
              </button>
            )}
          </div>
        </div>
      )}
    </div>
  );
};

export default SDSSection;

import React, { useState, useEffect } from 'react';

const SafetyComplianceTools = ({ sdsData, onValidationUpdate }) => {
  const [selectedStandards, setSelectedStandards] = useState(['OSHA 1910.1200']);
  const [jurisdictionChecks, setJurisdictionChecks] = useState({});
  const [hazardAnalysis, setHazardAnalysis] = useState({});

  const regulatoryStandards = [
    'OSHA 1910.1200 (US Hazard Communication)',
    'GHS (Globally Harmonized System)',
    'REACH (EU Registration, Evaluation, Authorization)',
    'WHMIS (Canada Workplace Hazardous Materials)',
    'Safe Work Australia'
  ];

  const jurisdictions = [
    { code: 'US', name: 'United States' },
    { code: 'EU', name: 'European Union' },
    { code: 'CA', name: 'Canada' },
    { code: 'AU', name: 'Australia' },
    { code: 'ZA', name: 'South Africa' },
    { code: 'GLOBAL', name: 'Global Standards' }
  ];

  const runComplianceCheck = async (standards, jurisdiction) => {
    try {
      const result = await safetyApi.checkSDSCompliance(sdsData.id, standards, jurisdiction);
      return result;
    } catch (error) {
      console.error('Compliance check failed:', error);
      return { compliant: false, issues: ['Check failed'] };
    }
  };

  const analyzeHazards = async (sdsContent) => {
    try {
      const analysis = await safetyApi.analyzeSDSHazards(sdsContent);
      setHazardAnalysis(analysis);
    } catch (error) {
      console.error('Hazard analysis failed:', error);
    }
  };

  useEffect(() => {
    if (sdsData) {
      analyzeHazards(sdsData);
    }
  }, [sdsData]);

  return (
    <div className="safety-compliance-tools">
      <h4>Safety Compliance Tools</h4>

      {/* Regulatory Standards Selection */}
      <div className="tool-section" style={{ marginBottom: '20px' }}>
        <h5>Regulatory Standards</h5>
        {regulatoryStandards.map(standard => (
          <label key={standard} style={{ display: 'block', marginBottom: '6px' }}>
            <input
              type="checkbox"
              checked={selectedStandards.includes(standard)}
              onChange={(e) => {
                if (e.target.checked) {
                  setSelectedStandards([...selectedStandards, standard]);
                } else {
                  setSelectedStandards(selectedStandards.filter(s => s !== standard));
                }
              }}
              style={{ marginRight: '8px' }}
            />
            {standard}
          </label>
        ))}
      </div>

      {/* Jurisdiction Validation */}
      <div className="tool-section" style={{ marginBottom: '20px' }}>
        <h5>Jurisdiction Validation</h5>
        {jurisdictions.map(jurisdiction => (
          <button
            key={jurisdiction.code}
            onClick={async () => {
              const result = await runComplianceCheck(selectedStandards, jurisdiction.code);
              setJurisdictionChecks(prev => ({
                ...prev,
                [jurisdiction.code]: result
              }));
            }}
            className="btn btn-sm btn-outline-primary"
            style={{ margin: '2px', fontSize: '12px' }}
          >
            {jurisdiction.name}
          </button>
        ))}

        {Object.entries(jurisdictionChecks).map(([code, result]) => (
          <div key={code} style={{ marginTop: '8px', fontSize: '12px' }}>
            <span style={{ fontWeight: '500' }}>
              {jurisdictions.find(j => j.code === code)?.name}:
            </span>
            <span style={{
              color: result.compliant ? '#059669' : '#dc2626',
              marginLeft: '6px'
            }}>
              {result.compliant ? '✓ Compliant' : '❌ Issues Found'}
            </span>
            {result.issues && result.issues.length > 0 && (
              <ul style={{ margin: '4px 0 0 20px', color: '#dc2626' }}>
                {result.issues.map((issue, idx) => (
                  <li key={idx}>{issue}</li>
                ))}
              </ul>
            )}
          </div>
        ))}
      </div>

      {/* Hazard Classification */}
      <div className="tool-section" style={{ marginBottom: '20px' }}>
        <h5>Hazard Analysis</h5>
        {hazardAnalysis.categories && (
          <div>
            {Object.entries(hazardAnalysis.categories).map(([category, present]) => (
              <div key={category} style={{ marginBottom: '4px' }}>
                <span style={{ fontWeight: '500' }}>{category}:</span>
                <span style={{
                  color: present ? '#dc2626' : '#059669',
                  marginLeft: '6px'
                }}>
                  {present ? '⚠️ Present' : '✓ Not Present'}
                </span>
              </div>
            ))}
          </div>
        )}
      </div>

      {/* PPE Requirements Validation */}
      <div className="tool-section">
        <h5>PPE Requirements Validation</h5>
        <p style={{ fontSize: '12px', color: '#6b7280', marginBottom: '8px' }}>
          Verify that PPE requirements in Section 8 match identified hazards
        </p>
        <button
          onClick={() => onValidationUpdate('ppe_check', true, 'PPE requirements validated')}
          className="btn btn-sm btn-success"
          style={{ width: '100%' }}
        >
          Validate PPE Requirements
        </button>
      </div>
    </div>
  );
};

  const [messages, setMessages] = useState([]);
  const [isTyping, setIsTyping] = useState(false);
  const messagesEndRef = useRef(null);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  useEffect(() => {
    const initialMessage = {

      sender: 'agent',
      content: `Welcome to SDS Review Assistant. I'm here to help you review and validate the Safety Data Sheet for ${context === 'sds_review' ? 'this procurement item' : 'the current task'}. I can answer questions about regulatory compliance, suggest revisions, or provide guidance on safety data sheet requirements.`,

      setIsTyping(false);

const resolveSDSHITL = async (taskId, resolution, feedback) => {
  switch (resolution) {
    case 'approved':
      await approveSDS(taskId);
      await updateOriginalProcurementTask(taskId, 'completed');
      break;

    case 'revision_requested':
      await requestSDSRevision(taskId, feedback);
      await createRevisionTask(taskId, feedback);
      break;

      await escalateSDSTask(taskId, feedback);
